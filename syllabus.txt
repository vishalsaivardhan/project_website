function selectYear(year) {
  currentYear = year;

  document.getElementById("semesterTabs").style.display = "flex";

  document.querySelectorAll(".year-block").forEach(y => y.style.display = "none");
  const y = document.querySelector(".year-" + year);
  if (y) y.style.display = "block";

  document.querySelectorAll(".sem").forEach(s => s.style.display = "none");
  document.querySelectorAll(".sem-" + year).forEach(s => {
    s.style.display = "inline-flex";
  });

  document.querySelectorAll(".sem-block").forEach(s => s.style.display = "none");
  document.querySelectorAll(".units").forEach(u => u.style.display = "none");
}

   SEMESTER SELECTION
====================== */
function selectSem(sem) {
  document.querySelectorAll(".sem-block").forEach(s => s.style.display = "none");
  const active = document.querySelector(".sem-" + sem);
  if (active) active.style.display = "block";
}

/* ======================
   SUBJECT CLICK
====================== */
document.addEventListener("click", e => {
  const subject = e.target.closest(".subject");
  if (!subject) return;

  // Year 3 & 4 ‚Üí info popup only
  if (currentYear === 3 || currentYear === 4) {
    showPopup("‚ö†Ô∏è Detailed syllabus will be updated soon üôÇ");
    return;
  }

  const units = subject.nextElementSibling;
  if (!units || !units.classList.contains("units")) return;

  document.querySelectorAll(".units").forEach(u => {
    if (u !== units) u.style.display = "none";
  });

  units.style.display =
    units.style.display === "block" ? "none" : "block";
});

/* ======================
   POPUP
====================== */
function showPopup(text) {
  const old = document.querySelector(".popup");
  if (old) old.remove();

  const popup = document.createElement("div");
  popup.className = "popup";
  popup.innerHTML = `<div class="popup-box">${text}</div>`;
  document.body.appendChild(popup);

  setTimeout(() => popup.remove(), 2500);
}
